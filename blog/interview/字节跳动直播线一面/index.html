类似于 forceUpdate 的自定义 Hook：

function useForceUpdate() {
  const [state, setState] = React.useState(false);

  return () => {
    setState(!state);
  };
}

// 递归实现
function flat_1(arr) {
  const result = [];

  for (const item of arr) {
    if (Array.isArray(item)) {
      result.push(...flat(item));
    } else {
      result.push(item);
    }
  }

  return result;
}

// 借助toString
function flat_2(arr) {
  return arr.toString().split(",").map(Number);
